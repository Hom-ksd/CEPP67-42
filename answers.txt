มีผลดี เพราะถือว่านายแดงไม่เคยต้องคำพิพากษาว่าได้กระทำความผิด
ได้ เพราะเป็นการกระทำที่กฎหมายให้ถือว่าเป็นการกระทำในราชอาณาจักร
ได้ เพราะการที่นายบุญมีลักทรัพย์นายโดมอนคนรัสเซีย ขณะที่เครื่องบินกำลังขึ้นจากสนามบินสุวรรณภูมิ ถือว่าเหตุเกิดบนพื้นอากาศเหนือราชอาณาจักร เป็นการกระทำความผิดในราชอาณาจักรตามประมวลกฎหมายอาญามาตรา 4 วรรคแรก จึงสามารถดำเนินคดีกับนายบุญมีได้
พิพากษายกฟ้องเพราะความผิดเกิดนอกราชอาณาจักร และทั้งผู้เสียหายและผู้กระทำความผิดเป็นคนต่างด้าว
ไม่มีอำนาจลงโทษเพราะไม่มีการกระทำส่วนหนึ่งส่วนใดเกิดขึ้นในเรือไทย กรณีตาม ป.อ. มาตรา 4 วรรคสอง ต้องมีการกระทำส่วนหนึ่งส่วนใดเกิดขึ้นในเรือไทย จึงให้ถือว่าเป็นการกระทำความผิดในราชอาณาจักร ส่วนกรณีตาม ป.อ. มาตรา 5 วรรคแรก ผลที่เกิดขึ้นในราชอาณาจักรไม่หมายความรวมถึงเรือไทยหรืออากาศยานไทยที่กฎหมายเพียงแต่ให้ถือว่าเป็นราชอาณาจักร
มีอำนาจลงโทษเพราะการตระเตรียมวางเพลิงมีกฎหมายบัญญัติให้เป็นความผิด และหากมีการกระทำต่อไปถึงขั้นความผิดสำเร็จ ผลจะเกิดขึ้นในประเทศไทย
นายอับดุลเข้ามาเผาบ้านนายบังเลาะที่อำเภอเบตง จังหวัดยะลาเป็นการกระทำความผิดในราชอาณาจักร จึงต้องรับโทษในราชอาณาจักรตาม ป.อ. มาตรา 4 วรรคแรก
ลงโทษนายอับดุลไม่ได้เพราะการกระทำของนายอับดุลอยู่เพียงขั้นตระเตรียมฆ่า และลงโทษนายฮากิมฐานเป็นผู้สนับสนุนไม่ได้เพราะความผิดยังมิได้กระทำลงในราชอาณาจักร
ศาลไทยลงโทษนายกิมย้งได้เพราะมีการกระทำผิดในเรือไทย ความผิดนอกราชอาณาจักรตาม ป.อ. มาตรา 7 ต้องมิได้เกิดในสถานที่ซึ่งถือว่าเป็นการกระทำความผิดในราชอาณาจักร นายกิมย้งปล้นเรือสินค้าไทย ซึ่งมาตรา 4 วรรคสองให้ถือว่าเป็นการกระทำความผิดในราชอาณาจักร ศาลไทยจึงมีอำนาจลงโทษตามหลักดินแดน
ลงโทษได้ เพราะการกระทำความผิดฐานชิงทรัพย์เริ่มเมื่อนายหนานให้ขนมผสมยากดประสาทแก่นายบุญตันกินในเขตประเทศลาว และมีการเอาทรัพย์ไปในเขตประเทศไทย ถือว่าการกระทำส่วนหนึ่งเกิดในราชอาณาจักร
ลงโทษไม่ได้ เพราะผลของการกระทำไม่ได้เกิดในราชอาณาจักรและไม่มีส่วนหนึ่งส่วนใดของการกระทำเกี่ยวข้องกับราชอาณาจักร
ความผิดฐานพรากผู้เยาว์อายุไม่เกิน 18 ปี เริ่มต้นตั้งแต่ที่นายเอกพานางสาวตรีขึ้นรถยนต์ที่หน้าปากซอยแล้ว แม้จะมีการร่วมประเวณีกันที่ประเทศญี่ปุ่น การกระทำของนายเอกเป็นการพรากผู้เยาว์เพื่อการอนาจาร ซึ่งมีการกระทำส่วนหนึ่งส่วนใดเกิดขึ้นในราชอาณาจักร นายโทบิดาของนางสาวตรีสามารถดำเนินคดี กับนายเอกได้ตาม ป.อ. มาตรา 5 วรรคแรก ฎีกาที่ 1645/2531
ศาลไทยลงโทษนายแซมตาม ป.อ. มาตรา 282 ได้ ตามหลักอำนาจลงโทษสากล ความผิดตาม ป.อ. มาตรา 282 เป็นความผิดที่ระบุไว้ในมาตรา 7 (2) ทวิ ดังนั้น แม้ว่าความผิดนี้จะไม่มีส่วนเกี่ยวข้องกับประเทศไทยเลย ศาลไทยก็มีอำนาจลงโทษได้ตามหลักกฎหมายมาตรา 7 จะต้องรับโทษในราชอาณาจักร
ดำเนินคดีกับนายแมสซี่ไม่ได้ เพราะการกระทำความผิดของนายแมสซี่มิได้กำหนดไว้ในมาตรา 8 ตาม ป.อ. มาตรา 8 การกระทำความผิดนอกราชอาณาจักร โดยผู้กระทำความผิดเป็นคนต่างด้าว และคนไทยเป็นผู้เสียหาย ซึ่งจะต้องรับโทษในราชอาณาจักรนั้น ต้องเป็นความผิดตามที่ระบุไว้ในมาตรา 8(1)-(13) ซึ่งความผิดที่ระบุไว้นั้นล้วนเป็นความผิดที่ผู้กระทำจะต้องกระทำโดยเจตนาทั้งสิ้น
ได้ เพราะนายมานพผู้เสียหายเป็นคนไทยและได้ร้องขอให้ลงโทษ ซึ่งเป็นกรณีการกระทำความผิดนอกราชอาณาจักรที่มีคนไทยเป็นผู้เสียหาย ตามข้อเท็จจริง เหตุเกิดขึ้นเหนือน่านฟ้าของประเทศจีน จึงไม่ใช่ราชอาณาจักรไทยตาม ป.อ. มาตรา 4 วรรคแรก และไม่ได้เกิดขึ้นในอากาศยานไทย จึงมิใช่กรณีตาม ป.อ. มาตรา 4 วรรคสอง อย่างไรก็ตาม ตาม ป.อ. มาตรา 8 (ข) บัญญัติว่า หากผู้เสียหายเป็นคนไทยและได้ร้องขอให้ลงโทษ และเป็นความผิดที่ระบุไว้ในมาตรา 8 ศาลไทยก็มีอำนาจลงโทษได้ เมื่อการกระทำของนายโตโต้เป็นความผิดฐานลักทรัพย์ ซึ่งเป็นความผิดที่ระบุไว้ในมาตรา 8 (8) ศาลไทยจึงมีอำนาจลงโทษนายโตโต้ได้ตาม ป.อ. มาตรา 8 (ข) (8).
ลงโทษไม่ได้ เพราการกระทำความผิดเกิดนอกราชอาณาจักร โดยคนไทยเป็นผู้เสียหายแต่ ผู้เสียหายไม่ได้ร้องขอให้ลงโทษ
ศาลไทยลงโทษนายเทพได้หากรัฐบาลสิงคโปร์ร้องขอให้ลงโทษ เพราะ แม้นายซ้งยังไม่ลงมือฆ่านายจิมมี่ นายเทพซึ่งเป็นผู้ใช้ต้องระวางโทษหนึ่งในสามของโทษในความผิดฐานฆ่าแล้ว ตามหลักกฎหมายมาตรา ๘๔ วรรคสอง เมื่อนายเทพกระทำความผิดนอกราชอาณาจักรจึงเป็นกรณีตาม ป.อ. มาตรา ๘ (ก) (๔) มิใช่กรณีตาม ป.อ. มาตรา ๖ เพราะความผิดที่ใช้ยังมิได้กระทำลงใน าชอาณาจักร
ลงโทษนายชาติฐานเป็นเจ้าพนักงานเรียกรับสินบนได้ตาม ป.อ. มาตรา ๙ แต่ลงโทษนายชอบไม่ได้ นายชาติกระทำความผิดเป็นเจ้าพนักงานเรียกหรือรับสินบนตาม ป.อ. มาตรา ๑๔๙ โดยมีนายชอบเป็นผู้สนับสนุน ศาลไทยมีอำนาจลงโทษนายชาติได้ตาม ป.อ. มาตรา ๙ แม้กระทำนอกราชอาณาจักร แต่นายชอบมิใช่เจ้าพนักงานจึงน ามาตรา ๙ มาใช้บังคับไม่ได้ และความผิดตาม ป.อ. มาตรา ๑๔๙ ไม่ใช่ความผิดที่กำหนดไว้ในมาตรา ๗ และ ๘ จึงลงโทษนายชอบในราชอาณาจักรไม่ได้ 
ลงโทษได้ การกระทำของนายเฮงเป็นความผิดตาม ป.อ. มาตรา ๒๔๐ และเป็นความผิดนอกราชอาณาจักรต้องรับโทษในราชอาณาจักรตาม ป.อ. มาตรา ๗ (๒) นายเฮงได้รับโทษจำคุกตามคำพิพากษาของศาลกัมพูชามาแล้ว แต่ยังไม่พ้นโทษ ศาลไทยจะลงโทษนายเฮงน้อยกว่าที่กฎหมายกำหนดไว้เพียงใดก็ได้ ทั้งนี้โดยคำนึงถึงโทษที่นายเฮงได้รับมาแล้ว ตาม ป.อ. มาตรา ๑๐ วรรคสอง 
การกระทำของนายโจเซฟและนายโชคเป็นความผิดตาม ป.อ. มาตรา ๒๔๐ ประกอบมาตรา ๒๔๗ ซึ่งเป็นการกระทำความผิดนอกราชอาณาจักรตาม ป.อ. มาตรา ๗(๒) แต่ได้มีค าพิพากษาถึงที่สุดให้ปล่อยนายโจเซฟแล้ว จึงลงโทษนายโจเซฟในประเทศไทยอีกไม่ได้ตาม ป.อ. มาตรา ๑๐(๑) ส่วนนายโชคยังไม่ได้รับโทษในความผิดนั้นมาก่อน จึงสามารถลงโทษนายโชคในประเทศไทยได้
ศาลไทยลงโทษนายวัน นายเดือนและนายปีได้ หากนายจันทร์บิดาของพระภิกษุรูปหนึ่งที่ถูกทั้ง สามคนฆ่าร้องขอให้ลงโทษ นายวัน นายเดือนและนายปีกระทำความผิดนอกราชอาณาจักรโดยที่ผู้เสียหายเป็นคนไทยและผู้เสียหายได้ร้องขอให้ลงโทษ เมื่อการกระทำของทั้งสามคนเป็นความผิดฐานปล้นทรัพย์และฆ่าผู้อื่น ตามที่กำหนดไว้ในมาตรา ๘ นายวัน นายเดือนและนายปีจึงต้องรับโทษในราชอาณาจักร 
เมื่อรัฐบาลไทยไม่ได้ร้องขอให้ลงโทษ ศาลไทยจึงลงโทษนายโชคได้อีก แต่จะลงโทษน้อยกว่าที่กฎหมายกำหนดไว้สำหรับความผิดฐานกระทำอนาจารเพียงใดก็ได้ นายเอกไม่ต้องรับผิดต่อนางตรีเพราะไม่มีการกระทำแต่ต้องรับผิดต่อนายโทเพราะเป็นการกระทำโดยประมาท 
การที่นายเอไม่จุดไฟหรือให้สัญญาณแก่รถคันอื่น ถือเป็นการงดเว้น นายเอผิดประมาทเป็นเหตุ ให้นายบีถึงแก่ความตาย 
นายแดง ผิดฐานฆ่าคนตายโดยประมาทเป็นการกระทำ โดยปราศจากความระมัดระวังตามวิสัยของคนใช้ปืน
ผิดฆ่านายขาวโดยเจตนาย่อมเล็งเห็นผล
นายแดงผิดพยายามฆ่านายดำและผิดฐานฆ่านายขาวตายโดยเจตนาย่อมเล็งเห็นผล
ไม่มีความผิดเพราะเป็นการป้องกันโดยสำคัญผิด ตามปัญหานางเงาะเข้าใจผิดว่านายกล้วยเป็นคนร้าย การกระทำของนางเงาะจึงเป็นการสำคัญผิดในข้อเท็จจริง ถ้าหากเป็นร้ายจะมาลักทรัพย์จริง ย่อมเป็นภยันตรายที่เกิดจากการประทุษร้ายอันละเมิดต่อกฎหมายและเป็นภยันตรายที่ใกล้จะถึง นางเงาะย่อมมีสิทธิป้องกันตนได้ การกระทำของนางเงาะจึงเป็นการป้องกันโดยสำคัญผิดในข้อเท็จจริง ย่อมไม่เป็นความผิดตาม ป.อ. มาตรา ๖๘ และมาตรา ๖๒
ผิดประมาทเป็นเหตุให้นายเขียดถึงแก่ความตาย เพราะความสำคัญผิดเกิดขึ้นด้วยความประมาท 
นายดำต้องรับผิดต่อนายขาวเพราะเป็นการกระทำโดยเจตนาแต่ไม่ต้องรับโทษหนักขึ้น ตาม ป.อาญา มาตรา ๖๑ จะยกข้อแก้ตัวว่าเป็นการสำคัญผิดในตัวบุคคลไม่ได้ นายดำจึงต้องรับผิดต่อนายขาว แต่ไม่ต้องรับโทษหนักขึ ้นเพราะกระทำต่อบุพการีเพราะนายดำไม่รู ้ข้อเท็จจริงนั ้นตาม ป.อ. มาตรา ๖๒ วรรคท้าย
เป็นการกระทำโดยป้องกันโดยชอบด้วยกฎหมายโดยสำคัญผิด จึงไม่มีความผิด ตาม ป.อ. มาตรา ๖๘ และมาตรา ๖๒
หากกระทำในขณะที่รู้ผิดชอบอยู่บ้าง ศาลจะลงโทษน้อยกว่าที่กฎหมายกำหนดก็ได ตาม ป.อ. มาตรา ๖๕ วรรคสอง 
กระทำโดยประมาทเป็นเหตุให้ผู้อื่นได้รับอันตรายสาหัส ตาม ป.อ. มาตรา ๖๖ จะยกเอาความมึนเมาเป็นข้อแก้ตัวว่ากระทำไปโดยไม่รู้ผิดชอบหรือไม่สามารถบังคับตนเองได้ เพื่อให้ตนเองพ้นผิดตาม ป.อ. มาตรา ๖๕ นั้นต้องเป็นกรณีที่ความมึนเมานั้นเกิดขึ้นโดยผู้เสพย์ไม่รู้ว่าสิ่งนั้นจะทำให้มึนเมาหรือถูกขืนใจให้เสพย์
ไม่มีความผิดทำให้เสียทรัพย์เพราะอ้างเหตุกระทำด้วยการป้องกัน เป็นกรณีที่เจ้าของช้างกระทำโดยประมาทในการควบคุมดูแลช้างที่กำลังตกมันซึ่งเป็นสัตว์ดุ ฎีกาที่ ๓๔๓๕/๒๕๒๗ เมื่อช้างนั้นจะทำร้ายนายโด่งจึงเป็นภยันตรายที่เกิดจากการประทุษร้ายอันละเมิดต่อกฎหมายและเป็นภยันตรายที่ใกล้จะถึง นายโด่งจึงป้องกันตนเองได้ เมื่อเป็นการป้องที่ชอบด้วยกฎหมายและพอสมควรแก่เหตุ นายโด่งจึงไม่มีความผิดตาม ป.อ. มาตรา ๖๘
ผิดฐานทำให้เสียทรัพย์แต่ไม่ต้องรับโทษ การที่นายเตี้ยยื่นมือเข้าไปในกรงเสือ แล้วถูกเสือตะปบมือได้รับบาดเจ็บนายเตี้ยจึงเป็นผู้ก่อให้เกิดภัยนั้นขึ้นก่อน จึงอ้างเหตุป้องกันมิได้ การที่นายโตใช้ปืนยิงเสือเพื่อช่วยนายเตี้ยจึงเป็นกระทำความผิดด้วยความจำเป็นเพื่อให้นายเตี ้ยพ้นจากภยันตรายซึ่งไม่สามารถหลีกเลี่ยงให้พ้นด้วยวิธีอื่นใดได้ และการที่เสือตะปบและกัดมือนายเตี้ยนั้นนายโตมิได้เป็นผู้ก่อให้เกิดภัยนั้นขึ้น ดังนั้น เมื่อนายโตยิงเสือตายเพื่อช่วยนายเตี้ย การกระทำของนายโตจึงเป็นการกระทำความผิดฐานทำให้เสียทรัพย์ แต่ไม่ต้องรับโทษเพราะกระทำความผิดด้วยความจำเป็นตาม ป.อ. มาตรา ๖๗ 
ไม่มีความผิดเพราะอ้างป้องกันได้ การทะเลาะวิวาทสิ้นสุดลงแล้ว นายแจ๊สยังตามมาจะแทงทำร้ายนายจิ้ม จึงเป็นภยันตรายซึ่งเกิดจากการประทุษร้ายอันละเมิดต่อกฎหมาย และเป็นภยันตรายที่ใกล้จะถึง นายจิ้มจึงมีสิทธิป้องกันตนเองได้ การที่นายจิ้มใช้มีดแทงสวนไปถูกนายแจ๊สถึงแก่ความตายจึงเป็นการป้องกันตนที่พอสมควรแก่เหตุ นายจิ้มจึงไม่มีความผิดตาม ป.อ. มาตรา ๖๘ 
มาตรา 59 วรรคหนึ่ง วรรคสอง วรรคสาม และวรรคสี่ “บุคคลจะต้องรับผิดในทางอาญา ก็ต่อเมื่อได้กระทําโดยเจตนา เว้นแต่จะได้กระทําโดยประมาท ในกรณีที่กฎหมายบัญญัติให้ต้องรับผิดเมื่อได้ กระทําโดยประมาท หรือเว้นแต่ในกรณีที่กฎหมายบัญญัติไว้โดยแจ้งชัดให้ต้องรับผิดแม้ได้กระทําโดยไม่มีเจตนา มาตรา 62 วรรคสอง “ถ้าความไม่รู้ข้อเท็จจริงตามความในวรรคสามแห่งมาตรา 59 หรือ ความสําคัญผิดว่ามีอยู่จริงตามความในวรรคหนึ่ง ได้เกิดขึ้นด้วยความประมาทของผู้กระทําความผิด ให้ผู้กระทํา รับผิดฐานกระทําโดยประมาท ในกรณีที่กฎหมายบัญญัติไว้โดยเฉพาะว่า การกระทํานั้นผู้กระทําจะต้องรับโทษ แม้กระทําโดยประมาท”
